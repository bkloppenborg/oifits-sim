00053     // Convert (North, East, Up) to (x,y,z) as defined by the APIS++ standards
00054     // see:    http://aips2.nrao.edu/docs/glossary
00055     // for more information.

00057     this->xyz[0] = -sin(phi) * NEU[0] + cos(phi) * NEU[2];
