00051 void FTPedretti::StoreOneFrame( Row<int>& detectorframe, double time )
00052 {
00053         // acquire fringes, detect potential problems (bad frames), convert to double precision
00054         
00055         for(int channel = 0 ; channel < nchannels ; channel++) frames[ channel ][ framecounter ] = (double)( detectorframe[ channel ] ) ; 
00056         
00057         // Storing data about the current frame
00058         frametime[ framecounter ] = (time + previous_time) / 2. ;
00059         frameduration[ framecounter ] = time - previous_time ;
00060         modulator_position[ framecounter ] = modulator->GetDelay( frametime[ framecounter ] );
00061         deltamod[ framecounter ] = modulator_position[ framecounter ] - modulator->GetDelay( previous_time );

00065         {
00066                 framecounter = 0 ; // reset frame counter for the next batch of coherent integration

00075 {
00076         
00077         for( int channel = 0 ; channel < nchannels ; channel++ )                
00078         {
00079                 ntilde[ channel ]  = ZEROCOMP;
00080                 
00081                 for( int framenumber = 0 ; framenumber < n_coh ; framenumber++ )
00082                 {
00083                         ntilde[ channel ] += frames[ channel ][ framenumber ] * exp( - 2. * PI * I * spectr->mean_wavenumber[ channel ] * modulator_position[ framenumber ] ) ; 
00084                 }
00085         }
00086         
00087         // The complex visibility is A + i B, with: 
00088         // A = C cos(D) + sin( C + D )
00089         // B = C sin(D) + cos( C + D ) 
00090         // C = 4*pi*wavenumber*scan_ampl*T
00091         // D = 2*pi*wavenumber*OPD
00092         // Computed with Maple:
00093         // f:=t->int(cos(2*Pi*s*(a*t+OPD))*exp(-2*I*Pi*s*a*t), t);
00094         // simplify(evalc(Im(f(T))/Re(f(T))));
00095         
00096         // Using arctan( B / A ) to get the phase only works under certain conditions
00097         // C >>> 1, preferably > 50, ensures this works adequately,
00098         // That means that the integration time should be T > 50/(4*pi*scan_ampl*wavenumber)
00099         // Then B/A = C cos(D)/C sin(D) = tan(D)
00100         // which gives OPD = atan(B/A) / ( 2*pi*wavenumber )
00101         // The Cross-spectra is also generally more precise for this reason (amongst others)
00102         // Note that this condition is independent of the value of the OPD
00103
