00065 // the values depend on: the visibility, the number of telescopes, the

00068 Matrix < Complex > Vis2Bis(Matrix < Complex > &visibility, int N, int Nwav,
00069    int Nbs, int Nha)
00070 {
00071         int i = 0;                                      // just a counter
00072         // declaring the bispectrum under the variable name bis; it is a
00073         // complex matrix
00074         Matrix < Complex > bis(Nwav, Nbs);
00075 
00076         for (int ii = 0; ii < Nwav; ii++)
00077         {
00078                 i = 0;
00079                 for (int jj = 0; jj < Nha; jj++)
00080                 {
00081                         for (int j = 0; j < N - 2; j++)
00082                         {
00083                                 for (int k = j + 1; k < N - 1; k++)
00084                                 {
00085                                         bis[ii][i] =
00086                                            visibility[ii][j +
00087                                            jj * N * (N - 1) / 2]
00088                                            * conj(visibility[ii]
00089                                            [k + jj * N * (N - 1) / 2])
00090                                            * visibility[ii][i + (N - 1) * (jj + 1)];
00091                                         i++;
00092                                 }
00093                         }
00094                 }
00095         }
00096         return (bis);

00299 // geocentric coordinates

00311 void wgs84_to_geoc(double lat, double height, double *GeocLat,
00312    double *GeocRadius)
00313 {
00314         double cosLat, sinLat, c, c0, s0, one_f_sq;
00315 
00316         cosLat = cos(lat);
00317         sinLat = sin(lat);
00318         one_f_sq = (1. - WGS_F) * (1. - WGS_F);
00319         c = pow(cosLat * cosLat + one_f_sq * sinLat * sinLat, -0.5);
00320         c0 = WGS_A0 * c + height;
00321         s0 = WGS_A0 * one_f_sq * c + height;
00322 
00323         *GeocLat = atan2(s0 * sinLat, c0 * cosLat);
00324         *GeocRadius =
00325            pow(c0 * c0 * cosLat * cosLat + s0 * s0 * sinLat * sinLat, 0.5);
00326 }
