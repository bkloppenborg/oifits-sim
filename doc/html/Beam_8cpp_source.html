00049 void Beam::ComputePhase(double time)
00050 // Get the atmosphere phase and apply the AO correction (at lambda0)
00051 // Add the modulator and delaylines delays
00052 {
00053   // Apply the mask to the phase screen
00054   phase = this->array->layer[station_index]->phase_screen * mask ;
00055 
00056   // computes the AO + tip/tilt correction
00057   if (AOorder > 0)
00058   {
00059     for (int ii = 1; ii <= AOorder; ii++)
00060     {
00061       phase -= zernlib->modes[ ii ] * scalprod(phase, zernlib->modes[ ii ]);
00062     }
00063   }
00064 
00065   // Now update the piston at lambda0
00066   // No high order here, but if need be the phase could be modified to introduce some
00067   // Note: we do not use phase += piston*zernike[0] as the OPD alone is needed in later computations
00068 
00069   double delay_modulator = this->modulator->GetDelay(time);
00070   double delay_delaylines = this->delayline->GetCurrentPosition(time);
00071 
00072   delay = delay_modulator + delay_delaylines;
