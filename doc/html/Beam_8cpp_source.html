00050 void Beam::ComputePhase(double time)
00051 // Get the atmosphere phase and apply the AO correction (at lambda0)
00052 // Add the modulator and delaylines delays
00053 {
00054   // Apply the mask to the phase screen
00055   phase = this->array->GetStation(station_index).layer->phase_screen * mask ;
00056 
00057   // computes the AO + tip/tilt correction
00058   if (AOorder > 0)
00059   {
00060     for (int ii = 1; ii <= AOorder; ii++)
00061     {
00062       phase -= zernlib->modes[ ii ] * scalprod(phase, zernlib->modes[ ii ]);
00063     }
00064   }
00065 
00066   // Now update the piston at lambda0
00067   // No high order here, but if need be the phase could be modified to introduce some
00068   // Note: we do not use phase += piston*zernike[0] as the OPD alone is needed in later computations
00069 
00070   double delay_modulator = this->modulator->GetDelay(time);
00071   double delay_delaylines = this->delayline->GetCurrentPosition(time);
00072 
00073   delay = delay_modulator + delay_delaylines;
00074 }
