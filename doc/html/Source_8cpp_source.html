00064 double Source::Spectrum(double wavenumber)
00065 {
00066     // return the black-body intensity in photons
00067     // Spectral radiance W.m-2.s-1.sr-1.m-1
00068     // double L = (2 * h * c * c / pow( wavelength , 5) / ( exp( h * c / (
00069     // wavelength * k * temperature ) -1.0)
00070     // Number of photons = L*wavelength/(h*c), nph.m-2.s-1.sr-1.m-1
00071     // double nph = L*wavelength/(h*c);
00072 
00073     // flux * nph / nph_max, where nph_max gives the maximum of emission
00074     return (flux + 0. * wavenumber);
00075 }

00082 void Source::InitFluxes(char band, double background_magnitude,
00083                         double sky_background_aperture)
00084 {
00085     // Flux density for a magnitude 0 star in nph.m.m-2.s-1
00086     // Multiply by waveband width (in wavenumber unit = m-1 ), collecting
00087     // surface, and integration time to get the number of photons
00088     // see e.g. Campins, Reike, & Lebovsky (1985) -- M0 fluxes in Jansky =
00089     // 1590, 1080, 667 for J, H , K
00090     double flux_m0 = 0.0;
00091 
00093     if (band == 'J')
00094         flux_m0 = 1.59e-23 / 6.626068e-34 * 1.235e-6;   // J band
00095     if (band == 'H')
00096         flux_m0 = 1.08e-23 / 6.626068e-34 * 1.635e-6;   // H band
00097     if (band == 'K')
00098         flux_m0 = 6.67e-24 / 6.626068e-34 * 2.159e-6;   // K band
00099     // Flux density for the chosen magnitude
00100     this->flux = flux_m0 * pow(10., -0.4 * magnitude);
00101 
00102     // Background noise calculations
00103     // The background magnitude is in magnitude/ sq arcsec and assumed given
00104     // in the same band as the source
00105     // The sky background aperture is in arcseconds on sky (MROI = 0.5 )
00106     this->backgroundflux = pow(sky_background_aperture, 2) * flux_m0 * pow(10.,-0.4 * background_magnitude);
00107     double SNR = flux / backgroundflux;
00108 
00109     cout << "Flux = " << this->flux << " photons per m, per second." << endl;
00110     cout << "Background flux = " << this->backgroundflux
00111         << " photons per m, per second." << endl;
00112     cout << "Source SNR: " << SNR << endl;
00113 }
