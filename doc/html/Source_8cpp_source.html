00102 double Source::Spectrum(double wavenumber)
00103 {
00104     // return the black-body intensity in photons
00105     // Spectral radiance W.m-2.s-1.sr-1.m-1
00106     // double L = (2 * h * c * c / pow( wavelength , 5) / ( exp( h * c / (
00107     // wavelength * k * temperature ) -1.0)
00108     // Number of photons = L*wavelength/(h*c), nph.m-2.s-1.sr-1.m-1
00109     // double nph = L*wavelength/(h*c);
00110 
00111     // flux * nph / nph_max, where nph_max gives the maximum of emission
00112     return (flux + 0. * wavenumber);

00120 void Source::InitFluxes(char band, double background_magnitude,
00121                         double sky_background_aperture)
00122 {
00123     // Flux density for a magnitude 0 star in nph.m.m-2.s-1
00124     // Multiply by waveband width (in wavenumber unit = m-1 ), collecting
00125     // surface, and integration time to get the number of photons
00126     // see e.g. Campins, Reike, & Lebovsky (1985) -- M0 fluxes in Jansky =
00127     // 1590, 1080, 667 for J, H , K
00128     double flux_m0 = 0.0;
00129 
00130     if (band == 'J')
00131         flux_m0 = 1.59e-23 / 6.626068e-34 * 1.235e-6;   // J band
00132     if (band == 'H')
00133         flux_m0 = 1.08e-23 / 6.626068e-34 * 1.635e-6;   // H band
00134     if (band == 'K')
00135         flux_m0 = 6.67e-24 / 6.626068e-34 * 2.159e-6;   // K band
00136     // Flux density for the chosen magnitude
00137     this->flux = flux_m0 * pow(10., -0.4 * magnitude);
00138 
00139     // Background noise calculations
00140     // The background magnitude is in magnitude/ sq arcsec and assumed given
00141     // in the same band as the source
00142     // The sky background aperture is in arcseconds on sky (MROI = 0.5 )
00143     this->backgroundflux = pow(sky_background_aperture, 2) * flux_m0 * pow(10.,-0.4 * background_magnitude);
00144     double SNR = flux / backgroundflux;
00145 
00146     cout << "Flux = " << this->flux << " photons per m, per second." << endl;
00147     cout << "Background flux = " << this->backgroundflux
00148         << " photons per m, per second." << endl;
00149     cout << "Source SNR: " << SNR << endl;
00150 }
