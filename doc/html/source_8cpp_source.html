00091     // Spectral radiance W.m-2.s-1.sr-1.m-1
00092     // double L = (2 * h * c * c / pow( wavelength , 5) / ( exp( h * c / (
00093     // wavelength * k * temperature ) -1.0)

00095     // double nph = L*wavelength/(h*c);
00096 
00097     // flux * nph / nph_max, where nph_max gives the maximum of emission
00098     return (flux + 0. * wavenumber);

00106 void Source::InitFluxes(char band, double background_magnitude,
00107                         double sky_background_aperture)
00108 {
00109     // Flux density for a magnitude 0 star in nph.m.m-2.s-1
00110     // Multiply by waveband width (in wavenumber unit = m-1 ), collecting
00111     // surface, and integration time to get the number of photons
00112     // see e.g. Campins, Reike, & Lebovsky (1985) -- M0 fluxes in Jansky =
00113     // 1590, 1080, 667 for J, H , K
00114     double flux_m0 = 0.0;
00115 
00116     if (band == 'J')
00117         flux_m0 = 1.59e-23 / 6.626068e-34 * 1.235e-6;   // J band
00118     if (band == 'H')
00119         flux_m0 = 1.08e-23 / 6.626068e-34 * 1.635e-6;   // H band
00120     if (band == 'K')
00121         flux_m0 = 6.67e-24 / 6.626068e-34 * 2.159e-6;   // K band
00122     // Flux density for the chosen magnitude
00123     this->flux = flux_m0 * pow(10., -0.4 * magnitude);
00124 
00125     // Background noise calculations
00126     // The background magnitude is in magnitude/ sq arcsec and assumed given
00127     // in the same band as the source
00128     // The sky background aperture is in arcseconds on sky (MROI = 0.5 )
00129     this->backgroundflux = pow(sky_background_aperture, 2) * flux_m0 * pow(10.,-0.4 * background_magnitude);
00130     double SNR = flux / backgroundflux;
00131 
00132     cout << "Flux = " << this->flux << " photons per m, per second." << endl;
00133     cout << "Background flux = " << this->backgroundflux
00134         << " photons per m, per second." << endl;
00135     cout << "Source SNR: " << SNR << endl;
00136 }
