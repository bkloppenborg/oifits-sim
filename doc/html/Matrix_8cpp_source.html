00213 // Function to interpolate a 2d rectangular grid onto another 2d
00214 // rectangular grid with arbitrary scaling, translation and
00215 // rotation. Uses bilinear interpolation and uses periodic boundary
00216 // conditions i.e. wraps the boundaries in the same way data from an
00217 // FFT is wrapped.
00218 // Taken from David Buscher and converted for use with this Matrix class by
00219 // Fabien Baron
00220 void Interp2d(Matrix < double >&dataIn, Matrix < double >&dataOut,
00221               double offsetX, double offsetY, double scale, double rotation)
00222 {
00223     // Matrix<double> &dataIn = Input grid of data points from which to
00224     // extract a subgrid. The grid pitch is assumed to be the same
00225     // in both dimensions (this restriction only holds if a non-zero rotation
00226     // is required)
00227     // Matrix<double> &dataOut = Output gridded values.
00228     // double offsetX = X offset of (0,0) point in output grid in input grid
00229     // coordinate units.
00230     // If greater than nxIn, the coordinates are wrapped in the X dimension.
00231     // double offsetY = Y offset of (0,0) point in output grid in input grid
00232     // coordinate units.
00233     // If greater than nyIn, the coordinates are wrapped in the Y dimension
00234     // double scale = output grid pitch measured in units of the input grid
00235     // pitch
00236     // double rotation = rotation between input and output grids in radians.
00237     // Positive values rotate the output axes *clockwise* wrt the input axes
00238 
00239     int ix0, ix1, iy0, iy1;
00240 
00241     double x, y, intX0, intX1, intY0, intY1, fracX, fracY;
00242 
00243     int nxIn = dataIn.GetCols();
00244 
00245     int nyIn = dataIn.GetRows();
00246 
00247     int nxOut = dataOut.GetCols();
00248 
00249     int nyOut = dataOut.GetRows();
00250 
00251     double c = scale * cos(rotation);
00252 
00253     double s = scale * sin(rotation);
00254 
00255     for (int iy = 0; iy < nyOut; iy++)
00256     {
00257         for (int ix = 0; ix < nxOut; ix++)
00258         {
00259             x = c * ix + s * iy + offsetX;
00260             y = -s * ix + c * iy + offsetY;
00261 
00262             /* Get integer and fractional parts of x position */
00263             intX0 = floor(x);
00264             intX1 = intX0 + 1.0;
00265             fracX = x - intX0;
00266 
00267             /* Do modulo arithmetic correctly for negative numbers */
00268             ix0 = (int)(intX0 - nxIn * floor(intX0 / nxIn));
00269             ix1 = (int)(intX1 - nxIn * floor(intX1 / nxIn));
00270 
00271             /* Repeat for y */
00272             intY0 = floor(y);
00273             intY1 = intY0 + 1.0;
00274             fracY = y - intY0;
00275             iy0 = (int)(intY0 - nyIn * floor(intY0 / nyIn));
00276             iy1 = (int)(intY1 - nyIn * floor(intY1 / nyIn));
00277 
00278             dataOut[ix][iy] = fracX * (fracY * dataIn[ix1][iy1] + (1.0 - fracY)
00279                                        * dataIn[ix1][iy0]) + (1.0 -
00280                                                               fracX) * (fracY *
00281                                                                         dataIn
00282                                                                         [ix0]
00283                                                                         [iy1] +
00284                                                                         (1.0 -
00285                                                                          fracY)
00286                                                                         *
00287                                                                         dataIn
00288                                                                         [ix0]
00289                                                                         [iy0]);
00290         }
