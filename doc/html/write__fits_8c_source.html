00047 char **make_tform(const char **template, int n, int value)
00048 {
00049         char **tform;
00050         int i, size, needed;
00051 
00052         tform = malloc(n * sizeof(char *));
00053         for (i = 0; i < n; i++)
00054         {
00055                 if (template[i][0] == '?')
00056                 {
00057                         size = strlen(template[i]) + 4; /* allow for extra digits
00058                                                                                          * + \0 */
00059                         tform[i] = malloc(size * sizeof(char));
00060                         needed =
00061                            snprintf(tform[i], size, "%d%s", value, &template[i][1]);
00062                         assert(needed < size);  /* fails if string was truncated */
00063                 }
00064                 else
00065                 {
00066                         tform[i] = malloc((strlen(template[i]) + 1) * sizeof(char));    /* \0 
00067                                                                                                                                                          */
00068                         strcpy(tform[i], template[i]);
00069                 }
00070         }
00071         return tform;
00072 }
