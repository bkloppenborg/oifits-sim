00033 UVPoint Baseline::UVcoords(double hour_angle, double source_declination, double wavenumber)
00034 {
00035     // First convert all values into radians (they should be degrees or decimal hours (of time) before now.
00036     double h = hour_angle * PI;
00037     double delta = source_declination;
00038 
00039     // Now compute the UV coordinates, again according to the APIS++ standards.
00040     UVPoint uv;
00041     uv.u = sin(h) * xyz[0]                    + cos(h) * xyz[1];
00042     uv.v = -sin(delta) * cos(h) * xyz[0]      + sin(delta) * sin(h) * xyz[1]    + cos(delta) * xyz[2];
00043     uv.w = cos(delta) * cos(h)*xyz[0]         - cos(delta) * sin(h) * xyz[1]    + sin(delta) * xyz[2];
00044     
00045     // Scale by the wavenumber (1/wavelength)
00048     uv.u *= wavenumber;
00049     uv.v *= wavenumber;
00050     uv.w *= wavenumber;
00051     
00052     return uv;
00053 }
